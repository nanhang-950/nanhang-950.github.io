路径和配置变量#

执行配置#

  • exec_tmout: 执行超时时间(毫秒)
  • hang_tmout: 挂起检测超时
  • mem_limit: 子进程内存限制(MB)
  • cpu_to_bind: CPU核心绑定

模糊测试标志#

  • skip_deterministic: 跳过确定性阶段
  • dumb_mode: 非插桩模式
  • crash_mode: 崩溃最小化模式
  • qemu_mode: QEMU模式
  • persistent_mode: 持久化模式

关键函数详解#

init_forkserver()#

功能: 初始化fork server 原理:

  • Fork server在目标程序初始化后启动
  • 通过管道与fuzzer通信
  • 每次测试快速fork子进程,避免重新加载程序

优势: 大幅提升执行速度(10-20倍)

  static struct itimerval it;

  int status;
  s32 rlen;

  if (pipe(st_pipe) || pipe(ctl_pipe)) PFATAL("pipe() failed");

has_new_bits()#

功能: 检测新的代码覆盖率 参数: virgin_map - 未触及位图 返回: 0=无新位, 1=有新位, 2=有新元组

工作流程:

  1. 比较trace_bits与virgin_map
  2. 检测是否有新路径被覆盖
  3. 更新virgin_map
static inline u8 has_new_bits(u8* virgin_map) {

#ifdef WORD_SIZE_64

  u64* current = (u64*)trace_bits;
  u64* virgin  = (u64*)virgin_map;

  u32  i = (MAP_SIZE >> 3);

#else

  u32* current = (u32*)trace_bits;
  u32* virgin  = (u32*)virgin_map;

  u32  i = (MAP_SIZE >> 2);

#endif /* ^WORD_SIZE_64 */

  u8   ret = 0;

  while (i--) {
    /* 检查当前执行是否触及了virgin_map中标记为未触及的位 */
    if (unlikely(*current) && unlikely(*current & *virgin)) {
      if (likely(ret < 2)) {
        
        u8* cur = (u8*)current;
        u8* vir = (u8*)virgin;

#ifdef WORD_SIZE_64

        /* 检查是否有完全未触及的字节(0xff) */
        if ((cur[0] && vir[0] == 0xff) || (cur[1] && vir[1] == 0xff) ||
            (cur[2] && vir[2] == 0xff) || (cur[3] && vir[3] == 0xff) ||
            (cur[4] && vir[4] == 0xff) || (cur[5] && vir[5] == 0xff) ||
            (cur[6] && vir[6] == 0xff) || (cur[7] && vir[7] == 0xff)) ret = 2;
        else ret = 1;

#else

        if ((cur[0] && vir[0] == 0xff) || (cur[1] && vir[1] == 0xff) ||
            (cur[2] && vir[2] == 0xff) || (cur[3] && vir[3] == 0xff)) ret = 2;
        else ret = 1;

#endif /* ^WORD_SIZE_64 */

      }
      /* 更新virgin_map,标记这些位已被触及 */
      *virgin &= ~*current;

    }
    current++;
    virgin++;
  }
  /* 如果发现新位且是主位图,标记位图已改变 */
  if (ret && virgin_map == virgin_bits) bitmap_changed = 1;

  return ret;
}

run_target()#

功能: 执行目标程序 流程:

  1. 清空trace_bits
  2. 通过fork server或直接fork执行
  3. 设置超时定时器
  4. 等待执行完成
  5. 分类计数并返回状态
static u8 run_target(char** argv, u32 timeout) {

  static struct itimerval it;
  static u32 prev_timed_out = 0;
  static u64 exec_ms = 0;

  int status = 0;
  u32 tb4;

  child_timed_out = 0;

  memset(trace_bits, 0, MAP_SIZE);
  MEM_BARRIER();

  if (dumb_mode == 1 || no_forkserver) {

    child_pid = fork();

    if (child_pid < 0) PFATAL("fork() failed");

    if (!child_pid) {

      struct rlimit r;

      if (mem_limit) {

        r.rlim_max = r.rlim_cur = ((rlim_t)mem_limit) << 20;

#ifdef RLIMIT_AS

        setrlimit(RLIMIT_AS, &r); /* Ignore errors */

#else

        setrlimit(RLIMIT_DATA, &r); /* Ignore errors */

#endif /* ^RLIMIT_AS */

      }

      r.rlim_max = r.rlim_cur = 0;

      setrlimit(RLIMIT_CORE, &r); /* Ignore errors */

      /* Isolate the process and configure standard descriptors. If out_file is
         specified, stdin is /dev/null; otherwise, out_fd is cloned instead. */

      setsid();

      dup2(dev_null_fd, 1);
      dup2(dev_null_fd, 2);

      if (out_file) {

        dup2(dev_null_fd, 0);

      } else {

        dup2(out_fd, 0);
        close(out_fd);

      }

      /* On Linux, would be faster to use O_CLOEXEC. Maybe TODO. */

      close(dev_null_fd);
      close(out_dir_fd);
      close(dev_urandom_fd);
      close(fileno(plot_file));

      /* Set sane defaults for ASAN if nothing else specified. */

      setenv("ASAN_OPTIONS", "abort_on_error=1:"
                             "detect_leaks=0:"
                             "symbolize=0:"
                             "allocator_may_return_null=1", 0);

      setenv("MSAN_OPTIONS", "exit_code=" STRINGIFY(MSAN_ERROR) ":"
                             "symbolize=0:"
                             "msan_track_origins=0", 0);

      execv(target_path, argv);

      *(u32*)trace_bits = EXEC_FAIL_SIG;
      exit(0);

    }

  } else {

    s32 res;

    if ((res = write(fsrv_ctl_fd, &prev_timed_out, 4)) != 4) {

      if (stop_soon) return 0;
      RPFATAL(res, "Unable to request new process from fork server (OOM?)");

    }

    if ((res = read(fsrv_st_fd, &child_pid, 4)) != 4) {

      if (stop_soon) return 0;
      RPFATAL(res, "Unable to request new process from fork server (OOM?)");

    }

    if (child_pid <= 0) FATAL("Fork server is misbehaving (OOM?)");

  }

  it.it_value.tv_sec = (timeout / 1000);
  it.it_value.tv_usec = (timeout % 1000) * 1000;

  setitimer(ITIMER_REAL, &it, NULL);

  if (dumb_mode == 1 || no_forkserver) {

    if (waitpid(child_pid, &status, 0) <= 0) PFATAL("waitpid() failed");

  } else {

    s32 res;

    if ((res = read(fsrv_st_fd, &status, 4)) != 4) {

      if (stop_soon) return 0;
      RPFATAL(res, "Unable to communicate with fork server (OOM?)");

    }

  }

4.4 calibrate_case()#

功能: 校准测试用例 目的:

  • 测量执行时间
  • 检测可变行为
  • 计算位图大小
  • 更新统计信息

fuzz_one()#

功能: 对一个测试用例进行完整的模糊测试 阶段:

  1. 位翻转(bitflip 1/1, 2/1, 4/1, 8/8, 16/8, 32/8)
  2. 算术运算(arith 8/16/32)
  3. 有趣值(interest 8/16/32)
  4. 字典(extras)
  5. Havoc(随机变异)
  6. Splicing(拼接)

十二、关键算法#

12.1 Effector Map#

识别对程序行为有影响的字节:

  1. 翻转每个字节
  2. 检查trace是否改变
  3. 标记有效字节

12.2 Token提取#

自动提取有趣的token:

  1. 在bitflip阶段观察
